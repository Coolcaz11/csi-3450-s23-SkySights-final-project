CREATE DATABASE IF NOT EXISTS final;

USE final;

CREATE TABLE EMPLOYEES (
	EMP_CODE INT PRIMARY KEY,
	EMP_FNAME TEXT,
	EMP_LNAME TEXT,
	CERT TEXT
)
ENGINE = INNODB;


CREATE TABLE RESULTS (
	TEST_CODE INT PRIMARY KEY,
	EMP_CODE INT,
	TEST_DESC TEXT,
	TEST_LAST_TEST_DONE DATE,
	TEST_NEXT_TEST_DUE DATE,
	TEST_RSLT TEXT,
	FOREIGN KEY (EMP_CODE) REFERENCES EMPLOYEES(EMP_CODE)
)
ENGINE = INNODB;



CREATE TABLE CUSTOMER (
	CUS_ACC_NUM INT PRIMARY KEY,
	CUS_FNAME TEXT,
	CUS_LNAME TEXT,
	CUS_BALANCE_DUE DECIMAL(10, 2),
	CUS_REQ TEXT
)
ENGINE = INNODB;


CREATE TABLE CHARTER_TRIP (
	CHARTER_NUM INT PRIMARY KEY,
	EMP_CODE INT,
	CUS_ACC_NUM INT,
	PILOT_CODE INT,
	CHARTER_TRIP_LENGTH TIME,
	CHARTER_COSTS DECIMAL(10, 2),
	CHARTER_REVENUE DECIMAL(10, 2),
	FOREIGN KEY (EMP_CODE) REFERENCES EMPLOYEES(EMP_CODE),
	FOREIGN KEY (CUS_ACC_NUM) REFERENCES CUSTOMER(CUS_ACC_NUM)
)
ENGINE = INNODB;
